Basic SQL Commands

SELECT

Retrieve data from database tables

SELECT column1, column2 FROM table_name;

SELECT ALL

Select all columns from a table

SELECT * FROM table_name;

WHERE

Filter records based on conditions

SELECT * FROM table_name WHERE condition;

ORDER BY

Sort results in ascending or descending order

SELECT * FROM table_name ORDER BY column_name ASC;

INSERT

Add new records to a table

INSERT INTO table_name (column1, column2) VALUES (value1, value2);

UPDATE

Modify existing records

UPDATE table_name SET column1 = value1 WHERE condition;

DELETE

Remove records from a table

DELETE FROM table_name WHERE condition;

DISTINCT

Return only unique values

SELECT DISTINCT column_name FROM table_name;

Advanced SQL Commands

GROUP BY

Group rows with same values

SELECT column1, COUNT(*) FROM table_name GROUP BY column1;

HAVING

Filter groups based on conditions

SELECT column1, COUNT(*) FROM table_name GROUP BY column1 HAVING COUNT(*) > 1;

LIMIT

Limit the number of results returned

SELECT * FROM table_name LIMIT 10;

SUBQUERY

Query within another query

SELECT * FROM table1 WHERE column1 IN (SELECT column1 FROM table2);

UNION

Combine results from multiple queries

SELECT column1 FROM table1 UNION SELECT column1 FROM table2;

CREATE TABLE

Create a new table

CREATE TABLE table_name (column1 datatype, column2 datatype);

ALTER TABLE

Modify table structure

ALTER TABLE table_name ADD column_name datatype;

DROP TABLE

Delete a table completely

DROP TABLE table_name;

SQL Functions

COUNT()

Count number of rows

SELECT COUNT(*) FROM table_name;

SUM()

Calculate sum of values

SELECT SUM(column_name) FROM table_name;

AVG()

Calculate average value

SELECT AVG(column_name) FROM table_name;

MAX()

Find maximum value

SELECT MAX(column_name) FROM table_name;

MIN()

Find minimum value

SELECT MIN(column_name) FROM table_name;

UPPER()

Convert text to uppercase

SELECT UPPER(column_name) FROM table_name;

LOWER()

Convert text to lowercase

SELECT LOWER(column_name) FROM table_name;

LENGTH()

Get length of text

SELECT LENGTH(column_name) FROM table_name;

SQL Joins

INNER JOIN

Return matching records from both tables

SELECT * FROM table1 INNER JOIN table2 ON table1.id = table2.id;

LEFT JOIN

Return all records from left table

SELECT * FROM table1 LEFT JOIN table2 ON table1.id = table2.id;

RIGHT JOIN

Return all records from right table

SELECT * FROM table1 RIGHT JOIN table2 ON table1.id = table2.id;

FULL OUTER JOIN

Return all records from both tables

SELECT * FROM table1 FULL OUTER JOIN table2 ON table1.id = table2.id;
